<script>
    const quotes = [
        "المتنبي: أعزُّ مكانٍ في الدُّنى سَرْجُ سابحٍ، وخيرُ جليسٍ في الأنامِ كتابُ.",

        "القراءة غذاء الروح ودواء العقل.",
        "أحمد شوقي: أنا من بَدّلَ بالكُتبِ الصِّحَابَا، لم أجدْ لي وافياً إلا الكِتَابَا."
    ];

    const students = {
        '1': ['أسامة المحاول', 'خالد جابر', 'سعد الجدوع'],
        '2': ['تركي الموينع', 'حسام الحقباني', 'راشد اليوسف'],
        '3': ['أحمد العمودي', 'أسامة الجطيل', 'خالد الحقباني']
    };

    const books = {
        '3': ['أبي كما عرفته', 'الرجل المهزوم', 'السجين يهرب', 'الوسائل المفيدة', 'لأنك الله متسلل'],
        '2': ['أبو هريرة', 'الرجال الثلاثة', 'القافلة', 'سوار أمي'],
        '1': ['أبو تراب', 'أبو عبيدة الجراح', 'أسد اليرموك', 'جامع القرآن']
    };

    document.getElementById('random-quote').innerText = quotes[Math.floor(Math.random() * quotes.length)];

    function showBooks(phase) {
        const screen = document.getElementById('books-screen');
        const list = document.getElementById('books-list');
        document.getElementById('phase-title').innerText = "كتب المرحلة: " + (phase == '1' ? 'الأولى' : phase == '2' ? 'الثانية' : 'الثالثة');
        
        list.innerHTML = '';
        books[phase].forEach(book => {
            let div = document.createElement('div');
            div.className = 'list-item';
            div.innerHTML = `<span>${book}</span><button onclick="openForm('${book}', '${phase}')" style="background: white; border:none; border-radius:5px; padding:5px 10px;">اختر</button>`;
            list.appendChild(div);
        });
        screen.style.display = 'block';
    }

    function openForm(book, phase) {
        document.getElementById('selected-book-title').innerText = book;
        const studentSelect = document.getElementById('student-name');
        studentSelect.innerHTML = '';
        students[phase].forEach(s => {
            let opt = document.createElement('option');
            opt.innerText = s;
            studentSelect.appendChild(opt);
        });
        document.getElementById('form-screen').style.display = 'block';
    }

    function closeScreen(id) { document.getElementById(id).style.display = 'none'; }

    function submitBook() {
        alert("تم حفظ البيانات بنجاح! شكراً لك يا بطل.");
        closeScreen('form-screen');
    }

    function checkAdmin() {
        let pass = prompt("كلمة مرور المالك:");
        if (pass === "w1234") {
            alert("مرحباً بك في لوحة الإحصائيات (قيد العمل)");
        } else {
            alert("تعذر الدخول، كلمة المرور خاطئة");
        }
    }
</script>
